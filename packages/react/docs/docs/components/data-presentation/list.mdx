### 尺寸

```jsx live
function Demo() {
  const items = [
    { label: '苹果' },
    { label: '香蕉' },
    { label: '橙子' },
    { label: '葡萄' },
  ]

  const [size, setSize] = useState('md')

  const sizes = ['xs', 'sm', 'md', 'lg', 'xl']
  return (
    <>
      <SpaceItems>
        {sizes.map(s => (
          <CRadio
            key={s}
            value={size}
            onChange={setSize}
            label={s}
            selectedValue={s}
          />
        ))}
      </SpaceItems>
      <CList size={size} items={items} itemKey="label" />
    </>
  )
}
```

### 激活项&可点击

```jsx live
function Demo() {
  const items = [
    { label: '苹果' },
    { label: '香蕉' },
    { label: '橙子' },
    { label: '葡萄' },
  ]

  const [activeFruit, setActiveFruit] = useState('苹果')

  return (
    <CList
      items={items}
      clickable
      itemKey="label"
      activeFn={item => item.label === activeFruit}
      onItemClick={item => setActiveFruit(item.label)}
    />
  )
}
```
